# Veritas Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a cross-institution federated fraud-intelligence demo that trains a real `flock-sdk`-style federated model across simulated banks and visualises siloed-vs-federated fraud spread in a live split-screen race.

**Architecture:** A Python `core/` service runs real federated learning (NumPy logistic model, FedAvg, Multi-Krum, differential privacy, two regimes) and exposes a REST + SSE contract via FastAPI. A Next.js `web/` app renders the split-screen race, population particle simulation, and counters, with a server-side MiniMax explainer. The two meet only at `contract/`; a standalone `mock/` server lets the web build start immediately.

**Tech Stack:** Python 3.11, NumPy, FastAPI, sse-starlette, pytest (core); Node/Express + SSE (mock); Next.js App Router, TypeScript, Tailwind, canvas, OpenAI SDK→MiniMax (web).

---

## Execution model

- **Phase 0** is done first by one person — it produces `contract/` + `mock/`.
- **Phase A** (`core/`, Person A) and **Phase B** (`web/`, Person B) then run **in parallel**. Person B builds entirely against `mock/`.
- **Phase C** integrates and polishes.

Event transport is **SSE** (`GET /events`, `text/event-stream`) — simpler and more robust than WebSocket for our one-way server→client stream. This refines the spec's "WebSocket" wording.

import numpy as np
def init_weights(dim): return np.zeros(dim+1)
def _aug(X): return np.hstack([X,np.ones((X.shape[0],1))])
def predict_proba(w,X): return 1.0/(1.0+np.exp(-np.clip(_aug(X)@w,-30,30)))
def train_local(w,X,y,epochs=10,lr=0.3,l2=1e-4):
    Xa=_aug(X); w=w.copy(); n=len(y)
    for _ in range(epochs):
        g=Xa.T@(1.0/(1.0+np.exp(-np.clip(Xa@w,-30,30)))-y)/n + l2*w; w-=lr*g
    return w
def recall(w,X,y,thr=0.5):
    if y.sum()==0: return 1.0
    pred=predict_proba(w,X)>=thr; return float((pred&(y==1)).sum()/y.sum())
```

import numpy as np
def fedavg(updates,weights=None):
    U=np.stack(updates)
    if weights is None: return U.mean(axis=0)
    wv=np.array(weights,float); wv/=wv.sum(); return (U*wv[:,None]).sum(axis=0)
def multi_krum(updates,n_byzantine,m):
    U=np.stack(updates); n=len(U); k=max(1,n-n_byzantine-2)
    dist=np.zeros((n,n))
    for i in range(n):
        for j in range(i+1,n):
            d=float(np.sum((U[i]-U[j])**2)); dist[i,j]=dist[j,i]=d
    scores=[np.sort(dist[i])[1:k+1].sum() for i in range(n)]
    sel=list(np.argsort(scores)[:m]); return U[sel].mean(axis=0), sel
```

import numpy as np
def clip_update(u,max_norm):
    nrm=float(np.linalg.norm(u)); return u if nrm<=max_norm else u*(max_norm/nrm)
def add_noise(u,sigma,max_norm,rng): return u+rng.normal(0,sigma*max_norm,size=u.shape)
def privatize(u,max_norm,sigma,rng): return add_noise(clip_update(u,max_norm),sigma,max_norm,rng)
```

from .dp import privatize
from .attack import poisoned_update

NAMES=["Barclays","NatWest","Lloyds","HSBC","Santander","Monzo","Starling","Nationwide"]

class Engine:

        self.use_dp=use_dp; self.use_krum=use_krum

        self.global_w=init_weights(FEATURE_DIM)

        self.eval=[make_bank_data(1000,0.05,seed=seed+100+i) for i in range(n_banks)]

    def inject_campaign(self):

        self.eval[0]=inject_campaign(*self.eval[0],80,seed=2); self.seen[0]=True
    def inject_attack(self,bank_id):

                self.data[i]=inject_campaign(*self.data[i],120,seed=10+i)
                self.eval[i]=inject_campaign(*self.eval[i],60,seed=20+i); self.seen[i]=True; break
    def step(self):

        if self.campaign: self._spread()
        ups=[]

            X,y=self.data[i]; wl=train_local(self.global_w,X,y,epochs=8,lr=0.3)

            ups.append(u)
        if self.attack_bank is not None and self.round<3:
            ups[self.attack_bank]=poisoned_update(ups[self.attack_bank],12.0)
        if self.use_krum:

        self.global_w=self.global_w+agg
        for i in range(self.n):
            X,y=self.data[i]; self.silo_w[i]=train_local(self.silo_w[i],X,y,epochs=8,lr=0.3)
        fed,silo=self._det()
        for i in range(self.n):
            ev.append({"type":"client_updated","data":{"bankId":f"bank{i}",
                "detection":{"federated":fed[i],"siloed":silo[i]}}})
        self.cum["fed"]+=sum(AT_RISK*(1-d) for d in fed)/self.n
        self.cum["silo"]+=sum(AT_RISK*(1-d) for d in silo)/self.n
        ev.append({"type":"round_complete","data":self.state()}); return ev
    def _det(self):
        fed=[recall(self.global_w,self.eval[i][0],self.eval[i][1]) for i in range(self.n)]
        silo=[recall(self.silo_w[i],self.eval[i][0],self.eval[i][1]) for i in range(self.n)]
        return fed,silo
    def state(self):
        fed,silo=self._det()
        banks=[{"id":f"bank{i}","name":NAMES[i],"customers":CUST[i],
            "detection":{"federated":round(fed[i],3),"siloed":round(silo[i],3)},
            "poisoned":self.attack_bank==i and self.round<3} for i in range(self.n)]
        fv=int(self.cum["fed"]); sv=int(self.cum["silo"])
        ttd=lambda a:HOURS*self.round if a>=THRESH else 101.0
        return {"round":self.round,"running":True,"banks":banks,
            "campaignActive":self.campaign,"attackActive":self.attack_bank is not None,
            "customerRecordsTransmitted":0,
            "counters":{
              "federated":{"fraudPreventedGbp":max(0,sv-fv)*AVG_LOSS,"timeToDetectHours":ttd(sum(fed)/self.n),"victims":fv,"lostGbp":fv*AVG_LOSS},
              "siloed":{"fraudPreventedGbp":0,"timeToDetectHours":ttd(sum(silo)/self.n),"victims":sv,"lostGbp":sv*AVG_LOSS}}}
```
